///Player_Tuta_Jump
if(global.pause) exit;
if(global.swap) exit;
if(global.shift) exit;
if(global.freeze2) exit;


if distance_to_object(WalljumpL) < 2 && place_free(x,y+1) {//tuta

vspeed=2;
maxFallSpeed=2;
if(global.player_head = 0){
    sprite_index=sprPlayerSliding;
    }
else{
    sprite_index=sprPlayerSlidingB;
    }
image_xscale=1
image_speed=1/2;


    if keyboard_check_pressed(global.rightbutton) && keyboard_check(global.jumpbutton) {
    if(global.player_head = 0){
        sprite_index=sprPlayerJump;
    }
    else{
        sprite_index=sprPlayerJumpB;
    }
    maxFallSpeed=8
    vspeed = -9;
    hspeed = 15;
    sound_play(snd_wallum);
    
    }
    
    if keyboard_check_pressed(global.rightbutton) && !keyboard_check(global.jumpbutton) {
    hspeed=3
    maxFallSpeed=8
    if(global.player_head = 0){
        sprite_index=sprPlayerFall;
    }
    else{
        sprite_index=sprPlayerFallB;
    }
    }

}

if distance_to_object(WalljumpL) > 1 && place_free(x,y+1) {
maxFallSpeed=8
}


if distance_to_object(WalljumpR) = 1 && place_free(x,y+1) {

vspeed=2;
maxFallSpeed=2;
if(global.player_head = 0){
    sprite_index=sprPlayerSliding;
    }
else{
    sprite_index=sprPlayerSlidingB;
    }
image_xscale=-1
image_speed=1/2;

    if keyboard_check_pressed(global.leftbutton) && keyboard_check(global.jumpbutton) {
    if(global.player_head = 0){
        sprite_index=sprPlayerJump;
    }
    else{
        sprite_index=sprPlayerJumpB;
    }
    maxFallSpeed=8
    vspeed = -9;
    hspeed = -15;
    sound_play(snd_wallum);
    }
    
    if keyboard_check_pressed(global.leftbutton) && !keyboard_check(global.jumpbutton) {
    hspeed=-3
    maxFallSpeed=8
    if(global.player_head = 0){
        sprite_index=sprPlayerFall;
    }
    else{
        sprite_index=sprPlayerFallB;
    }
    }

}

if distance_to_object(WalljumpR) > 1 && place_free(x,y+1) {
maxFallSpeed=8
}


if(place_meeting(x-5,y,WalljumpL) || place_meeting(x+5,y,WalljumpR)){
    tuta=1;
    }
else{
    tuta=0;
    }
