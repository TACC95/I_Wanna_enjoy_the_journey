///paused gravity

if(global.pause=1){
    tmp_hspeed=hspeed;
    tmp_vspeed=vspeed;
    tmp_imspeed=image_speed;
    hspeed=0;
    vspeed=0;
    image_speed=0;
    gravity=0;
    pause_flag=1;
    }
else if(pause_flag=1){
    gravity=0.4;
    hspeed=tmp_hspeed;
    vspeed=tmp_vspeed;
    image_speed=tmp_imspeed;
    pause_flag=0;
    }

//swap
if(global.swap=1){
    tmp_hspeed=hspeed;
    tmp_vspeed=vspeed;
    tmp_imspeed=image_speed;
    hspeed=0;
    vspeed=0;
    image_speed=0;
    gravity=0;
    swap_flag=1;
    }
else if(swap_flag=1){
    gravity=0.4;
    hspeed=tmp_hspeed;
    vspeed=tmp_vspeed;
    image_speed=tmp_imspeed;
    swap_flag=0;
    }

//shift
if(global.shift=1){
    tmp_hspeed=hspeed;
    tmp_vspeed=vspeed;
    tmp_imspeed=image_speed;
    hspeed=0;
    vspeed=0;
    image_speed=0;
    gravity=0;
    tmp_xscale=image_xscale;
    shift_flag=1;
    }
else if(shift_flag=1){
    gravity=0.4;
    hspeed=tmp_hspeed;
    vspeed=tmp_vspeed;
    image_speed=tmp_imspeed;
    image_xscale=tmp_xscale;
    shift_flag=0;
    }

//freeze2
if(global.freeze2=1){
    tmp_hspeed=hspeed;
    tmp_vspeed=vspeed;
    tmp_imspeed=image_speed;
    //tmp_gravity=gravity;
    hspeed=0;
    vspeed=0;
    image_speed=0;
    gravity=0;
    freeze2_flag=1
    }
else if(freeze2_flag=1){
    gravity=0.4;
    //gravity=tmp_gravity;
    hspeed=tmp_hspeed;
    vspeed=tmp_vspeed;
    image_speed=tmp_imspeed;
    freeze2_flag=0;
    }
